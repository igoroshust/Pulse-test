describe('Заполнение формы на стороннем сервисе', () => {
  it('должен заполнить данные в форме', () => {

    // Открыть страницу входа
    cy.visit('https://example.com/login'); // Замените на URL вашей страницы входа

    // Ввести учетные данные
    cy.get('input[name="username"]').type('yourUsername'); // Замените на правильный селектор
    cy.get('input[name="password"]').type('yourPassword'); // Замените на правильный селектор

    // Отправить форму входа
    cy.get('form').submit();

    // Подождать, пока загрузится страница после входа
    cy.url().should('include', '/dashboard'); // Замените на URL вашей страницы после входа

    // Теперь открываем страницу с формой
    cy.visit('https://example.com/form'); // Замените на URL вашего сервиса

    // Заполнить текстовые поля
    cy.get('input[name="username"]').type('testuser'); // Замените на правильный селектор
    cy.get('input[name="email"]').type('testuser@example.com'); // Замените на правильный селектор

    // Выбрать дату из календаря
    cy.get('input[name="date"]').click(); // Замените на правильный селектор
    cy.get('.calendar-day').contains('15').click(); // Замените на правильный селектор

    // Отправить форму
    cy.get('form').submit(); // Замените на правильный селектор

    // Проверить, что данные были успешно отправлены
    cy.url().should('include', '/success'); // Замените на правильный URL после успешной отправки
  });
});
